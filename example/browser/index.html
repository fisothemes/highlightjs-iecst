IEC-ST Highlight.js Demo

This is a demo of syntax highlighting for IEC 61131-3 Structured Text using Highlight.js + iecst grammar.

Function Block

// Controls a motor.
{attribute 'no_explicit_call' := 'do not call this POU directly'}
FUNCTION_BLOCK FB_Motor EXTENDS FB_Equipment IMPLEMENTS I_Motor
VAR
    _fSpeed      : LREAL := 100.5; (* RPM *)
    _bIsRunning  : BOOL;
END_VAR
METHOD Start : BOOL
    Start := THIS^._bIsRunning := TRUE;
END_METHOD
END_FUNCTION_BLOCK

Program

PROGRAM MAIN
VAR
    fbMotor : FB_Motor;
    bStart  : BOOL := TRUE;
END_VAR

IF bStart THEN
    fbMotor.Start();
END_IF
